{{- $url := .Destination -}} {{- $imageResource := partial "image-resource" (dict "Destination" $url "Page" .Page) -}} {{- if and (strings.HasPrefix $url "http") (not (strings.HasPrefix $url site.BaseURL)) -}} {{- /* Absolute links pointing to external pages, e.g. `https://example.com` */ -}} {{- .Text -}}{{- partial "icon" "arrow_outward" -}} {{- else if strings.HasPrefix $url "dialog:" -}} {{- /* Dialog triggers, e.g. `dialog:test` */ -}} {{- .Text -}}{{- partial "icon" "open_in_browser" -}} {{- else if strings.HasPrefix $url "mailto:" -}} {{- /* Email links, e.g. `mailto:example@example.com` */ -}} {{- partial "icon" "mail" -}}{{- .Text -}} {{- else if strings.HasPrefix $url "tel:" -}} {{- /* Telephone links, e.g. `tel:+1234567890` */ -}} {{- partial "icon" "call" -}}{{- .Text -}} {{- else if and $imageResource (strings.HasPrefix $imageResource.MediaType.Type "image") -}} {{- /* Internal image resources, e.g. `/operator/cd/file.png` */ -}} {{- $lightboxID := printf "lightbox-%s" ($imageResource.RelPermalink | sha1) -}} {{- .Text -}}{{- partial "icon" "image" -}} {{- partial "lightbox" (dict "ID" $lightboxID "Image" $imageResource "Alt" .Text) -}} {{- else if .Page.Ref (dict "path" $url) -}} {{- /* Internal links, referenced by path (e.g. `/news/1`) */ -}} {{- .Text -}} {{- else if or (strings.HasPrefix $url "/") (strings.HasPrefix $url site.BaseURL) -}} {{- /* Internal links, referenced by URL (e.g. `/en/news/1`) */ -}} {{- .Text -}} {{- else -}} {{- warnf "Link '%s' could not be resolved." .Destination -}} {{- end -}}
